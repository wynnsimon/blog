# 信号灯循环

```js
// 信号灯循环
function task(color, time) {
  return new Promise((resolve, reject) => {
    return setTimeout(() => {
      console.log(color);
      resolve();
    }, time);
  });
}

async function runTask() {
  await task("red", 2000);
  await task("yellow", 1000);
  await task("green", 3000);
  runTask();
}

runTask();
```
